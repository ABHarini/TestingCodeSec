sub hp()

Dim sh As Worksheet
Dim foundrow As Long
Dim cell As Range
Dim lastrow As Long
Dim row As Range
Dim yellowrow As Long


For Each sh In ThisWorkbook.Worksheets
For Each cell In Range("N2:N300")

 If cell.Value > 10000 Then
   foundrow = cell.row
   Rows.EntireRow(foundrow).Interior.Color = RGB(255, 255, 0)
 End If

Next cell
Next sh

Set sh = Sheets.Add
sh.Name = "Analysis"
lastrow = ThisWorkbook.Worksheets("Analysis").Range("A" & Rows.Count).End(xlUp).row

For Each sh In ThisWorkbook.Worksheets
For Each row In sh.UsedRange
   
     If Rows.Interior.Color = vbYellow Then
      yellowrow = row.row
      Rows.EntireRow(yellowrow).Copy , [Destination:ThisWorkbook.Worksheets("Analysis").Range (lastrow + 1)]
      'If cell.DisplayFormat.Interior.Color = vbYellow Then MainWs.Range(cell.Address) = cell.Value
     End If
Next row
Next sh

End Sub
